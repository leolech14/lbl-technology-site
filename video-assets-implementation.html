Performance Optimization

// Lazy load videos on intersection
const videoObserver = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
        if (entry.isIntersecting) {
            const video = entry.target.querySelector('video');
            video.play();
        } else {
            const video = entry.target.querySelector('video');
            video.pause();
        }
    });
});

document.querySelectorAll('.video-icon').forEach(icon => {
    videoObserver.observe(icon);
});

// Preload critical videos
<link rel="preload" as="video" href="assets/videos/hero-loop.webm">

Video Compression Settings

# Convert to WebM with transparency (VP9)
ffmpeg -i input.mov -c:v libvpx-vp9 -pix_fmt yuva420p -b:v 0 -crf 30 output.webm

# Convert to MP4 with H.265
ffmpeg -i input.mov -c:v libx265 -crf 28 -preset fast -tag:v hvc1 output.mp4

# Generate poster frame
ffmpeg -i video.webm -ss 00:00:00 -vframes 1 poster.jpg
